# Smaller build while keeping sound (QtMultimedia remains enabled):
# - prunes unneeded Qt runtime blocks (QML/WebEngine/etc)
# - keeps Spin/Ding assets and QSoundEffect
# - bundles OCR engine from owpicker_mvc/OCR/ (if present)
#   expected layout example:
#   owpicker_mvc/OCR/tesseract.exe
#   owpicker_mvc/OCR/*.dll
#   owpicker_mvc/OCR/tessdata/*.traineddata
set OW_BUILD_PROFILE=full
set OW_PRUNE_QT=1
set OW_INCLUDE_REQUESTS=1
set OW_INCLUDE_OCR_BUNDLE=1
pyinstaller --noconfirm --clean owpicker_mvc/OverwatchWheels.spec

# Minimal-size profile:
# - excludes QtMultimedia from bundle
# - does not embed Spin/Ding assets (falls back to QApplication.beep())
set OW_BUILD_PROFILE=minsize
pyinstaller --noconfirm --clean owpicker_mvc/OverwatchWheels.spec
